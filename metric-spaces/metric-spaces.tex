    A \textit{metric} of a \ex{Sets}{set} $M$ is a \ex{Functions}{function} $d : M \times M \to \R$ such that ($\forall x,y,z \in M$)
    \begin{enumerate}
        \item $d(x,y) = 0 \leftrightarrow x = y$,
        \item $d(x, y) = d(y, x)$
        \item $d(x,z) \le d(x,y) + d(y,z)$ \hspace{1em} (\ex{TriangleInequality}{triangle inequality})
    \end{enumerate}

    \begin{definition}[Metric Space]
        The pair $\left( M,d \right)$ is called a \textbf{metric space}.
    \end{definition}

    \begin{theorem} [Positivity] \label{positivity}
        $\forall x, y \in M, d(x,y) \ge 0$
    \end{theorem}
    \begin{proof}
        \phantom{}\vspace{-2em}
        \begin{align*}
            \forall x,y \in M,\hspace{1em} 0 &= d(x,x) && \text{A1} \\
            &\le d(x,y) + d(y,x) && \text{A3} \\
            &= 2d(x,y) && \text{A2} \\
            \to \hspace{1em} 0 &\le d(x,y). \qed
        \end{align*}
    \end{proof}
