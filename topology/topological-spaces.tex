    \pagestyle{plain} \currentdoc{note} \noindent
    A \textit{topology} $\mathcal{T}$ of a \ex{Sets}{set} $X$ is a \ex{Collections}{collection} of \ex{Subsets}{subsets} of $X$ such that
    \begin{enumerate}
        \item $\emptyset \in \mathcal{T}$ and $X \in \mathcal{T}$,
        \item $\mathcal{T}$ is closed under finite \ex{UnionsAndIntersections}{intersections}, and
        \item $\mathcal{T}$ is closed under arbitrary \ex{UnionsAndIntersections}{unions}.
    \end{enumerate}

    \begin{definition}[Topological Space]
        The pair $\left( X,\mathcal{T} \right)$ is called a \textbf{topological space}.
    \end{definition}

    \begin{definition}[Open Sets]
        A \ex{Subsets}{subset} $S \subseteq X$ of a \textbf{topological space} $\left( X,\mathcal{T} \right)$ is an \textbf{open set} iff $S \in \mathcal{T}$.
    \end{definition}
